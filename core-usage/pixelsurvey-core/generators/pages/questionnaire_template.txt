# Callback to enable/disable continue button based on all questionnaire questions being answered
@callback(
    [Output('$questionnaire_button_id$', 'disabled'),
     Output('$questionnaire_button_id$', 'href')],
     State('hrefs-store', 'data'),
    [Input(f'question-{i}', 'value') for i in range(1, $num_questionnaire_questions$ + 1)],
    prevent_initial_call=True
)
def toggle_continue_button(hrefs, *question_values):
    # Check if all questions have been answered (not None and not empty)
    all_answered = all(val is not None and val != "" for val in question_values)
    is_disabled = not all_answered
    next = hrefs['next']['/$activity_key$']
    href = f"{next}#top" if all_answered else None
    return is_disabled, href


# Callback to handle questionnaire button click - register responses and time
@callback(
    Output('$questionnaire_button_id$', 'n_clicks'),
    Input('$questionnaire_button_id$', 'n_clicks'),
    [State('hrefs-store', 'data'),
     State('user-store', 'data')] + [State(f'question-{i}', 'value') for i in range(1, $num_questionnaire_questions$ + 1)],
    prevent_initial_call=True
)
def handle_questionnaire_click(n_clicks, hrefs, user_data, *question_values):
    if n_clicks and user_data:
        responses_dict = {f'$activity_key$_q{i+1}': value for i, value in enumerate(question_values)}
        
        # Register responses and time in database
        db.register_questionnaire_responses(user_data['id'], '$activity_key$', responses_dict)
        db.register_time(user_data['id'], '$questionnaire_button_id$', time.time())

        if hrefs['next']['/$activity_key$'] == '/end':
            db.update_quota(user_data['id'])
    
    return n_clicks
